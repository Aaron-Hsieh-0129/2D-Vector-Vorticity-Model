212double vvm::Init::GetTB(int k, vvm &model) {
213 double z_top = 12000., T_top = 213., tb_top = 343.;
214 double z_t = model.dz * (k - 0.5);
215 if (z_t <= z_top) return 300. + 43. * pow(z_t / z_top, 1.25);

412void vvm::Init::RandomPerturbation(vvm &model, int t) {
413 std::mt19937 gen(t); // Mersenne Twister engine for random numbers
414 std::normal_distribution<> distribution(0.0, 1.0); // Gaussian distribution with mean 0 and standard deviation 1
415
416 // Parameters for the 2D Gaussian noise array
417 double mean = 0.; // Mean of the Gaussian distribution
418 double standard_deviation = 1.; // Standard deviation of the Gaussian distribution
419 double min_range = -0.25; // Minimum value of the generated noise
420 double max_range = 0.25; // Maximum value of the generated noise
421
422 double z = 0;
423
424 for (int k = 1; k < model.nz-1; k++) {
425 z = (k - 0.5) * model.dz;
426 for (int i = 1; i < model.nx-1; i++) {
427 if (z < 200) {
428 double random_noise = 0.;
429 do {
430 random_noise = mean + standard_deviation * distribution(gen);
431 } while (random_noise < min_range || random_noise > max_range);
432
433 model.init_th_forcing[i][k] = random_noise;
434 }
435 else {
436 model.init_th_forcing[i][k] = 0.;
437 }
438 }
439 }
