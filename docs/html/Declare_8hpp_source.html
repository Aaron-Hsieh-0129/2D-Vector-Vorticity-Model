227 static double** allocate2DContinuousArray(int rows, int cols, double*& contMemory) {
228 double** array = new double*[rows];
229 contMemory = new double[rows * cols]; // Allocate continuous memory block
230 for (int i = 0; i < rows; ++i) {
231 array[i] = &contMemory[i * cols]; // Point to segments within continuous block
232 }
233 return array;
234 }

237 if (array != nullptr) {
238 delete[] contMemory; // Deallocate the continuous block of memory
239 delete[] array; // Deallocate the array of pointers
240 }
241 }
