5void vvm::Turbulence::RKM_RKH(vvm &model) {
6 double Rzeta = 0.;
7 double Rotat = 0.;
8 double Ri = 0.;
9
10 #ifdef _OPENMP
11 #pragma omp parallel for collapse(2)
12 #endif
13 for (int k = 1; k <= model.nz-2; k++) {
14 for (int i = 1; i <= model.nx-2; i++) {
15 Rzeta = 0.5*((model.w[i+1][k+1] + model.w[i+1][k]) - (model.w[i-1][k+1] + model.w[i-1][k])) * model.r2dx +
16 0.5*((model.u[i+1][k+1] + model.u[i][k+1]) - (model.u[i+1][k-1] + model.u[i][k-1])) * model.r2dz;
17
18 Rotat = std::pow((model.u[i+1][k] - model.u[i][k]) * model.rdx, 2) + std::pow((model.w[i][k+1] - model.w[i][k]) * model.rdz, 2);
19
20 Ri = (model.GRAVITY / model.thb[k] * (model.thp[i][k+1] - model.thp[i][k-1]) * model.r2dz) / (std::pow(Rzeta, 2) + 2. * Rotat);
21
22 if (Ri < 0) {
23 model.RKM[i][k] = model.lambda2[k] * std::sqrt(std::pow(Rzeta, 2) + 2. * Rotat) * std::sqrt(1. - 16. * Ri);
24 model.RKH[i][k] = model.lambda2[k] * std::sqrt(std::pow(Rzeta, 2) + 2. * Rotat) * 1.4 * std::sqrt(1. - 40.*Ri);
25 }
26 else if (0 < Ri && Ri < 0.25) {
27 model.RKM[i][k] = model.lambda2[k] * std::sqrt(std::pow(Rzeta, 2) + 2. * Rotat) * std::pow(1. - 4.*Ri, 4);
28 model.RKH[i][k] = model.lambda2[k] * std::sqrt(std::pow(Rzeta, 2) + 2. * Rotat) * 1.4 * (1.-1.2*Ri)*std::pow(1.-4.*Ri, 4);
29 }
30 else {
31 model.RKM[i][k] = 0.;
32 model.RKH[i][k] = 0.;
33 }
34
35 // Diffusion should be larger than 1
36 model.RKM[i][k] = std::max(model.RKM[i][k], 1.);
37 model.RKH[i][k] = std::max(model.RKH[i][k], 1.);
38
39 // Diffusion should be smaller than 0.8 * dx^2 / dt
40 model.RKM[i][k] = std::min(model.RKM[i][k], 0.8 * model.dx * model.dz / model.dt);
41 model.RKH[i][k] = std::min(model.RKH[i][k], 0.8 * model.dx * model.dz / model.dt);
42 }
43 }
44 model.BoundaryProcess2D_center(model.RKM, model.nx, model.nz);
45 model.BoundaryProcess2D_center(model.RKH, model.nx, model.nz);
46
47 vvm::Turbulence::Mparam(model, model.zeta, model.zetap);
48 vvm::Turbulence::Hparam(model, model.th, model.thp);
49 #if defined(WATER)
50 vvm::Turbulence::Hparam(model, model.qv, model.qvp);
51 vvm::Turbulence::Hparam(model, model.qc, model.qcp);
52 vvm::Turbulence::Hparam(model, model.qr, model.qrp);
53 #endif

55 // std::cout << "min RKM: " << *std::min_element(model.RKMcont, model.RKMcont + model.nx*model.nz) << " min RKH: " << *std::min_element(model.RKHcont, model.RKHcont + model.nx*model.nz) << std::endl;
